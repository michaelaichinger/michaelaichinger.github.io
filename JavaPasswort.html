2.4 Java Passwort

Es wird ein zufälliges Passwort mit 4 Stellen generiert.
Anschließend wird das Passwort geknackt, indem alle möglichen Kombinationen durchprobiert werden.

import java.util.Random;
public class AufgabePasswort {

    public static void main(String[] args) {

        StringBuilder Password=new StringBuilder("    ");
        
        Random random=new Random();

        String pwchars="1234567890ßqwertzuiopü+asdfghjklöä#yxcvbnm,.-!§$%&/()=?QWERTZUIOPÜ*ASDFGHJKLÖÄ'YXCVBNM;:_";
        
        int randomInt=random.nextInt(pwchars.length());
        char pw1=pwchars.charAt(randomInt);
        Password.setCharAt(0, pw1);
        
        randomInt=random.nextInt(pwchars.length());
        char pw2=pwchars.charAt(randomInt);
        Password.setCharAt(1, pw2);
        
        randomInt=random.nextInt(pwchars.length());
        char pw3=pwchars.charAt(randomInt);
        Password.setCharAt(2, pw3);
        
        randomInt=random.nextInt(pwchars.length());
        char pw4=pwchars.charAt(randomInt);
        Password.setCharAt(3, pw4);
        
        int pwint=pwchars.length();
        StringBuilder password=new StringBuilder("    ");
        char pwc=' ';
        int l=1;
        int k=1;
        int j=1;
        int i=1;
        
        while (!(Password.toString().equals(password.toString()))) {
            while (pwint>=l&&k==pwint+1)
            {pwc=pwchars.charAt(pwint-l);
            password.setCharAt(0,pwc);
            System.out.println(password);
            l++;k=1;}
                while (pwint>=k&&j==pwint+1)
                {pwc=pwchars.charAt(pwint-k);
                password.setCharAt(1,pwc);
                System.out.println(password);
                k++;j=1;}
                    while (pwint>=j&&i==pwint+1)
                    {pwc=pwchars.charAt(pwint-j);
                    password.setCharAt(2,pwc);
                    System.out.println(password);
                    j++;i=1;}
                    while (pwint>=i&&!(Password.toString().equals(password.toString())))
                        {pwc=pwchars.charAt(pwint-i);
                        password.setCharAt(3,pwc);
                        System.out.println(password);
                        i++;}
                    }
        
        System.out.println("Das Passwort ist: "+password);
                
    }

}
