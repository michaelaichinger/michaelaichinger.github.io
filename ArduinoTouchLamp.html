3.5 Touch Lamp

Das Licht leuchtet, wenn man die Alufolie berührt.


#include <CapacitiveSensor.h>
    CapacitiveSensor capSensor = CapacitiveSensor(4,2);     //(4,2) = (Pin der das leitfähige Material sendet, Pin der das Material empfängt)
    
    int threshold = 50;       //Ab diesen Wert leuchtet das LED; Wenn der Wert nur knapp über den Standardwert der vom Sensor abgelesen wird ist, kann man das LED zum leuchten bringen indem man nur in die Nähe des Metalles kommt, ohne es zu berühren
    const int ledPin = 12;
    
    void setup() {
        Serial.begin(9600);
        pinMode(ledPin, OUTPUT);
    }
    
    void loop() {
        long sensorValue = capSensor.capacitiveSensor(30);    //Der Wert in der Klammer (x) ist die Anzahl der Proben die der Sensor abliest, bei zu wenig kann es zu Ungenauigkeiten kommen, bei zu vielen kann es zu Verzögerungen kommen
        Serial.println(sensorValue);
    
        if(sensorValue > threshold) {
        digitalWrite(ledPin, HIGH);
        }
        else {
        digitalWrite(ledPin, LOW);
        }
        delay(10);
    }
