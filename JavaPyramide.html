2.1 Java Pyramide

Man gibt ein wie hoch die Pyramide sein soll, dann wird sie mit dieser Höhe generiert.

import java.util.Scanner;

public class AufgabePyramide {

    public static void main(String[] args) {
        
        byte pyramide = -1;
        
        do {
            try {
                Scanner scanner = new Scanner(System.in);
                System.out.println("Wie hoch soll die Pyramide sein?");
                pyramide = scanner.nextByte();
            } catch (Exception e) {
                System.out.println("Nur gültige Zahl eingeben (max.127)");
            };
        } while (pyramide == -1);
        
        int j=pyramide;
        int k=0;
        int i=0;
        
        
        while (pyramide>0) {
            
        while (i<pyramide) 
            {System.out.print("  ");
            i++;
            }
        
        while (j-pyramide>=k)
        {System.out.print("X ");
        k++;
        }
        
        System.out.println();
        pyramide--; j++; k=0; i=0;
        }
        
    }

}
