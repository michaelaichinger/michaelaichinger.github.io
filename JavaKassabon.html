2.3 Java Kassabon

Die Kosten der Produkte werden ausgerechnet und wie ein Kassabon angezeigt.
Preis und Anzahl der Produkte können im Code geändert werden.

public class AufgabeKassabon {

	public static void main(String[] args) {
		double Wurst=4.20;
		double Kaese=2.30;
		double Brot=2.11;
		double DVD=12.00;
		double Geld=50.00;
		int Stk;
		
		Stk=4;
		double ergWurst=Wurst*Stk;
		System.out.println("Wurst	"+Stk+"x	"+Wurst+" EUR");
		System.out.println("				"+ergWurst+" EUR");
		
		Stk=1;
		double ergKaese=Kaese*Stk;
		System.out.println("Käse	"+Stk+"x	"+Kaese+" EUR");
		System.out.println("				"+ergKaese+" EUR");
		
		Stk=2;
		double ergBrot=Brot*Stk;
		System.out.println("Brot	"+Stk+"x	"+Brot+" EUR");
		System.out.println("				"+ergBrot+" EUR");
		
		Stk=3;
		double ergDVD=DVD*Stk;
		System.out.println("DVD	"+Stk+"x	"+DVD+" EUR");
		System.out.println("				"+ergDVD+" EUR");
		
		System.out.println("-----------------------------------------");
		
		double erg=ergWurst+ergKaese+ergBrot+ergDVD;
		System.out.println("Gesammt				"+erg+" EUR");
		
		System.out.println("Gegeben				"+Geld+" EUR");
		
		double Rueckgeld=Geld-erg;
		double Fehlgeld=erg-Geld;
		if (Geld>=erg) {System.out.println("Zurueck				"+Rueckgeld+" EUR");}
		else {System.out.println("Sie haben "+Fehlgeld+" EUR zu wenig!");}
				

	}

}
